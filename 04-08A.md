看到您遇到了 IjkPlayer 播放错误（what=-10000, extra=0），这个错误码通常表示网络连接失败或服务器拒绝访问。从截图看，部分电台（如湖南交通广播）可以播放，但 CMG环球资讯广播 等无法播放，这说明是具体流地址的问题。

一、播放错误分析
错误码	含义	常见原因
-10000	MEDIA_ERROR_UNKNOWN	网络超时、URL 无效、服务器返回非 200、缺少 User-Agent、HLS 流未正确配置
您的代码中，CMG环球资讯广播 的 URL 是 https://sk.cri.cn/905.m3u8，这是一个 HLS（m3u8） 流。IjkPlayer 默认支持 HLS，但需要正确设置 User-Agent 和 Referer，否则某些 CDN 会拒绝。

fun playStation(station: Station, retryWithSoftDecode: Boolean = false) {
    Log.d(TAG, "Playing station: ${station.name}, URL: ${station.url}")
    currentStation = station
    _playbackState.postValue(PlaybackState.Buffering)

    try {
        ijkPlayer?.let { player ->
            player.reset()
            // 如果是重试，临时禁用硬解码
            if (retryWithSoftDecode) {
                player.setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "mediacodec", 0)
                Log.d(TAG, "Retrying with software decoding")
            } else {
                player.setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "mediacodec", if (hardwareDecodeEnabled) 1 else 0)
            }
            player.configurePlayer()
            player.setupListeners()
            player.dataSource = station.url
            player.prepareAsync()
        }
    } catch (e: Exception) {
        Log.e(TAG, "Error playing station", e)
        _playbackState.postValue(PlaybackState.Error("播放失败: ${e.message}"))
    }
}

private inner class OnErrorListenerImpl : IMediaPlayer.OnErrorListener {
    override fun onError(mp: IMediaPlayer?, what: Int, extra: Int): Boolean {
        // 如果是 -10000 错误且还未重试过，则切换软解重试
        if (what == -10000 && !retried) {
            retried = true
            currentStation?.let { playStation(it, retryWithSoftDecode = true) }
            return true
        }
        val errorMsg = when (what) {
            IMediaPlayer.MEDIA_ERROR_NOT_VALID_FOR_PROGRESSIVE_PLAYBACK -> "流格式无效"
            IMediaPlayer.MEDIA_ERROR_UNKNOWN -> "未知错误 (网络或服务器拒绝)"
            else -> "播放错误: what=$what, extra=$extra"
        }
        Log.e(TAG, "Playback error: $errorMsg")
        _playbackState.postValue(PlaybackState.Error(errorMsg))
        stopPositionUpdates()
        return true
    }
}

package com.example.ijkradio.utils

import android.content.Context
import com.example.ijkradio.data.Station
import com.google.gson.Gson
import com.google.gson.reflect.TypeToken
import java.io.InputStreamReader

object JsonStationLoader {
    fun loadFromAssets(context: Context, fileName: String = "stations.json"): List<Station> {
        return try {
            val inputStream = context.assets.open(fileName)
            val reader = InputStreamReader(inputStream)
            val type = object : TypeToken<List<Station>>() {}.type
            val stations: List<Station> = Gson().fromJson(reader, type)
            reader.close()
            stations
        } catch (e: Exception) {
            e.printStackTrace()
            emptyList()
        }
    }
}

// 完整代码较长，我将关键修改部分贴出，您可直接用下面的完整版替换

package com.example.ijkradio.player

import android.content.Context
import android.os.Handler
import android.os.Looper
import android.util.Log
import androidx.lifecycle.LiveData
import androidx.lifecycle.MutableLiveData
import com.example.ijkradio.data.Station
import com.example.ijkradio.ui.PlaybackState
import tv.danmaku.ijk.media.player.IjkMediaPlayer
import tv.danmaku.ijk.media.player.IMediaPlayer

class IjkPlayerManager private constructor(private val context: Context) {

    companion object {
        private const val TAG = "IjkPlayerManager"
        @Volatile private var instance: IjkPlayerManager? = null
        fun getInstance(context: Context): IjkPlayerManager {
            return instance ?: synchronized(this) {
                instance ?: IjkPlayerManager(context.applicationContext).also { instance = it }
            }
        }
    }

    private var ijkPlayer: IjkMediaPlayer? = null
    private var currentStation: Station? = null
    private val _playbackState = MutableLiveData<PlaybackState>(PlaybackState.Stopped)
    val playbackState: LiveData<PlaybackState> = _playbackState
    private var isInitialized = false
    private var currentVolume = 1.0f
    private var hardwareDecodeEnabled = true
    private var retried = false

    fun initialize() {
        if (isInitialized) return
        try {
            IjkMediaPlayer.loadLibrariesOnce(null)
            ijkPlayer = IjkMediaPlayer().apply {
                configurePlayer()
                setupListeners()
            }
            isInitialized = true
        } catch (e: Exception) {
            _playbackState.postValue(PlaybackState.Error("播放器初始化失败: ${e.message}"))
        }
    }

    private fun IjkMediaPlayer.configurePlayer() {
        setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "mediacodec", if (hardwareDecodeEnabled) 1 else 0)
        setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "opensles", 1)
        setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "start-on-prepared", 1)
        setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "reconnect", 5)
        setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "max-buffer-size", 1024 * 1024)
        setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "fflags", "fastseek")
        setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "infbuf", 1)

        // 关键：设置 HTTP 头
        setOption(IjkMediaPlayer.OPT_CATEGORY_FORMAT, "user_agent", "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36")
        setOption(IjkMediaPlayer.OPT_CATEGORY_FORMAT, "referer", "https://sk.cri.cn/")
        setOption(IjkMediaPlayer.OPT_CATEGORY_FORMAT, "timeout", 30000000L)
        setOption(IjkMediaPlayer.OPT_CATEGORY_FORMAT, "connect_timeout", 30000000L)
    }

    private fun IjkMediaPlayer.setupListeners() {
        setOnPreparedListener {
            setVolume(currentVolume, currentVolume)
            start()
            currentStation?.let { _playbackState.postValue(PlaybackState.Playing(it.name)) }
        }
        setOnErrorListener { _, what, extra ->
            if (what == -10000 && !retried) {
                retried = true
                currentStation?.let { playStation(it, retryWithSoftDecode = true) }
                return@setOnErrorListener true
            }
            val errorMsg = if (what == IMediaPlayer.MEDIA_ERROR_UNKNOWN) "网络连接失败或服务器拒绝访问" else "播放错误($what,$extra)"
            _playbackState.postValue(PlaybackState.Error(errorMsg))
            true
        }
        setOnCompletionListener {
            _playbackState.postValue(PlaybackState.Stopped)
        }
        setOnInfoListener { _, what, _ ->
            when (what) {
                IMediaPlayer.MEDIA_INFO_BUFFERING_START -> _playbackState.postValue(PlaybackState.Buffering)
                IMediaPlayer.MEDIA_INFO_BUFFERING_END -> currentStation?.let { _playbackState.postValue(PlaybackState.Playing(it.name)) }
            }
            false
        }
    }

    fun playStation(station: Station, retryWithSoftDecode: Boolean = false) {
        retried = false
        currentStation = station
        _playbackState.postValue(PlaybackState.Buffering)
        try {
            ijkPlayer?.apply {
                reset()
                if (retryWithSoftDecode) {
                    setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "mediacodec", 0)
                } else {
                    setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "mediacodec", if (hardwareDecodeEnabled) 1 else 0)
                }
                configurePlayer()
                setupListeners()
                dataSource = station.url
                prepareAsync()
            }
        } catch (e: Exception) {
            _playbackState.postValue(PlaybackState.Error("播放失败: ${e.message}"))
        }
    }

    fun pause() { ijkPlayer?.pause(); _playbackState.postValue(PlaybackState.Paused) }
    fun resume() { ijkPlayer?.start(); currentStation?.let { _playbackState.postValue(PlaybackState.Playing(it.name)) } }
    fun stop() { ijkPlayer?.stop(); _playbackState.postValue(PlaybackState.Stopped) }
    fun setVolume(volume: Float) { currentVolume = volume.coerceIn(0f,1f); ijkPlayer?.setVolume(currentVolume, currentVolume) }
    fun isPlaying(): Boolean = ijkPlayer?.isPlaying() == true
    fun getCurrentStation(): Station? = currentStation
    fun release() { ijkPlayer?.release(); ijkPlayer = null; isInitialized = false }
}
